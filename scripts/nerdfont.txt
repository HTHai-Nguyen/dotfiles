# Nerd font 
echo
echo "Install Nerd font"
FONT_NAME="JetBrainsMono"
FONT_DIR="$HOME/.local/share/fonts/jetbrains-mono"
FONT_ZIP="$HOME/Download/JetBrainsMono.zip"

if fc-list | grep -i "$FONT_NAME" >/dev/null 2>&1; then
    echo "✅ Font '$FONT_NAME' is already installed. Skipping download and installation."
else
    # 2. Check if the font directory already exists
    if [ -d "$FONT_DIR" ]; then
        echo "✅ Font directory already exists: $FONT_DIR"
    else
        # If the directory does not exist, create it
        echo "Creating font directory: $FONT_DIR"
        mkdir -p "$FONT_DIR"
    fi

    # 3. Define the URL for the font (replace this URL with the latest version if available)
    FONT_URL="https://github.com/ryanoasis/nerd-fonts/releases/latest/download/JetBrainsMono.zip"

    # 4. Check if the font zip file already exists in the Download folder
    if [ -f "$FONT_ZIP" ]; then
        echo "✅ Font zip file already exists: $FONT_ZIP"
    else
        # If the font zip file doesn't exist, download it
        echo "Downloading JetBrainsMono.zip..."
        wget -O "$FONT_ZIP" "$FONT_URL"
    fi

    # 5. Extract the font to the font directory
    echo "Extracting font to: $FONT_DIR"
    unzip -o "$FONT_ZIP" -d "$FONT_DIR"

    # 6. Update the font cache
    echo "Updating font cache..."
    fc-cache -f -v

    echo "✅ JetBrains Mono Nerd Font setup complete!"
fi
